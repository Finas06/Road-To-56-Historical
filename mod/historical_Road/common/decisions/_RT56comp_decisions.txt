rt56_comp_winter_war = {
	winter_war_ongoing_mission = {
		activation = {
			 any_country = {
				has_country_flag = rt56_comp_winter_war_decisions_active
			}
		}
		available = {
			always = no
		}
		cancel_trigger = {
			FIN = {
				NOT = {
					has_war_with = SOV
				}
			}
		}
		custom_effect_tooltip = desc_winter_war
		is_good = yes
		days_mission_timeout = 120
		timeout_effect = {
			FIN = {
				white_peace = SOV
			}
			if = {
				limit = {#Finland Wins
					FIN = {
						controls_province = 3151 #Leningrad
						controls_province = 11105 #Helsinki
						controls_province = 9206 #Viipuri
					}
				}
				SOV = {
					add_timed_idea = {
						idea = rt56_comp_sov_reparations_winter_war
						days = 90
					}
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				FIN = {
					add_manpower = 75000
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				custom_effect_tooltip = rt56_comp_finnish_victory
			}
			else_if = {
				limit = {#Partial Finland Victory
					AND = {
						FIN = {
							controls_province = 11105 #Helsinki
						}
						OR = {
							FIN = {
								casualties < 50000
							}
							FIN = {
								controls_province = 9206 #Viipuri
							}
						}
					}
				}
				SOV = {
					transfer_state = 146 #Karjala
					transfer_state = 147 #Sala
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				FIN = {
					clr_country_flag = rt56_comp_winter_war_decisions_active
					add_manpower = 50000
				}
				custom_effect_tooltip = rt56_comp_slight_finnish_victory
			}
			else = {
				SOV = {#Soviet Victory
					transfer_state = 146 #Karjala
					transfer_state = 147 #Sala
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				FIN = {
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				custom_effect_tooltip = rt56_comp_soviet_victory
			}
		}
		complete_effect = { #SOV wins
			hidden_effect = {
				SOV = {
					transfer_state = 146 #Karjala
					transfer_state = 148 #Sala
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				FIN = {
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
			}
		}
		cancel_effect = { #Same as complete
			hidden_effect = {
				SOV = {
					transfer_state = 146 #Karjala
					transfer_state = 147 #Sala
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
				FIN = {
					clr_country_flag = rt56_comp_winter_war_decisions_active
				}
			}
		}
	}
	rt56_comp_SOV_winter_war_capitulation = {
